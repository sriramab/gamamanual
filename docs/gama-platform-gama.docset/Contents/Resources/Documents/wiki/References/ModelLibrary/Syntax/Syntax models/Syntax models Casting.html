model Casting

species to_int {
    init {
        write sample(int(1));
        write sample(int(1.0));
        write sample(int("1"));
        write sample(int("1.0"));
        write sample(int(°pink));
        write sample(int(true));
        write sample(int(self));
        write sample(int([]));
        write sample(int([0]));
        write sample(int({0,0,0}));
        write sample(int(0::0));
        write sample(int([0::0]));
        write sample(int(#AABBCC));
        write sample(int(°meter));              
    }
}
